--1A
Select E.F_NAME, E.L_NAME, J.START_DATE
from EMPLOYEES E
INNER JOIN JOB_HISTORY J ON E.EMP_ID = J.EMPL_ID
where E.DEP_ID = 5;

--1B
SELECT E.F_NAME, E.L_NAME, JH.START_DATE, J.JOB_TITLE
FROM JOB_HISTORY JH
INNER JOIN EMPLOYEES E ON E.EMP_ID = JH.EMPL_ID
INNER JOIN JOBS J ON JH.JOBS_ID = J.JOB_IDENT
WHERE E.DEP_ID=5;

--2A
SELECT  E.EMP_ID, E.L_NAME, E.DEP_ID, D.DEP_NAME
FROM EMPLOYEES E
LEFT JOIN DEPARTMENTS D
	ON E.DEP_ID = D.DEPT_ID_DEP;


--2B
SELECT  E.EMP_ID, E.L_NAME, E.DEP_ID, D.DEP_NAME
FROM EMPLOYEES E
LEFT JOIN DEPARTMENTS D
	ON E.DEP_ID = D.DEPT_ID_DEP
WHERE YEAR(E.B_DATE) <1980;

--2C

SELECT  E.EMP_ID, E.L_NAME, E.DEP_ID, D.DEP_NAME
FROM EMPLOYEES E
LEFT JOIN DEPARTMENTS D
	ON E.DEP_ID = D.DEPT_ID_DEP AND YEAR(E.B_DATE) <1980;

--3A

SELECT E.F_NAME, E.L_NAME, D.DEP_NAME
FROM EMPLOYEES E
FULL JOIN DEPARTMENTS D 
ON E.DEP_ID = D.DEPT_ID_DEP ORDER BY D.DEP_NAME, E.L_NAME;

--3B
SELECT E.F_NAME, E.L_NAME, D.DEP_NAME
FROM EMPLOYEES E
FULL JOIN DEPARTMENTS D 
ON E.DEP_ID = D.DEPT_ID_DEP AND E.SEX='M';
